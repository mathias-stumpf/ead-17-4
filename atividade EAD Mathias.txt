CREATE TABLE clientes (
    id INTEGER PRIMARY KEY,
    nome TEXT,
    email TEXT,
    cidade TEXT
);

CREATE TABLE produtos (
    id INTEGER PRIMARY KEY,
    nome TEXT,
    preco REAL,
    estoque INTEGER
);

CREATE TABLE pedidos (
    id INTEGER PRIMARY KEY,
    cliente_id INTEGER,
    produto_id INTEGER,
    quantidade INTEGER
);

SELECT * 
FROM produtos
WHERE preco > 1000;
SELECT *
FROM produtos

SELECT * FROM pedidos;
SELECT nome, email
FROM clientes;
SELECT 
    C.nome,
    COUNT(P.id) AS quantidade_pedidos
FROM clientes C
LEFT JOIN pedidos P ON C.id = P.cliente_id
GROUP BY C.nome;
SELECT 
    C.nome AS cliente,
    PR.nome AS produto,
    P.quantidade
FROM pedidos P
JOIN clientes C ON P.cliente_id = C.id
JOIN produtos PR ON P.produto_id = PR.id;
SELECT 
    C.nome,
    SUM(PR.preco * P.quantidade) AS total_gasto
FROM pedidos P
JOIN clientes C ON P.cliente_id = C.id
JOIN produtos PR ON P.produto_id = PR.id
GROUP BY C.nome;
SELECT 
    PR.nome
FROM produtos PR
LEFT JOIN pedidos P ON PR.id = P.produto_id
WHERE P.id IS NULL;
SELECT 
    C.nome,
    COUNT(P.id) AS total_pedidos
FROM pedidos P
JOIN clientes C ON P.cliente_id = C.id
GROUP BY C.nome
ORDER BY total_pedidos DESC
LIMIT 1;
